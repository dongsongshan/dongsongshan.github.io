wp-cli官网

wp-cli命令行的安装 
// 下载
curl -O https://raw.githubusercontent.com/wp-cli/builds/gh-pages/phar/wp-cli.phar

// 检查是否可以使用
php wp-cli.phar --info

// 更改权限，并移动到环境变量，改名为wp
chmod +x wp-cli.phar
sudo mv wp-cli.phar /usr/local/bin/wp

//下载wordpress
wp core download

//安装wordpress
wp core install

//查看当前wordpress版本
wp core version 

//检查wordpress，是否有可更新版本
wp core check-update

//列出插件列表
wp plugin list
