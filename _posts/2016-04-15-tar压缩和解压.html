tar压缩

//tar cf 压缩文件名.tar 要压缩的文件
tar cf dss.tar dong

//或者
//tar czf 压缩文件名.tar.gz 要压缩的文件
// c: 表示create; x:表示解压; z:表示gzip压缩; f:表示file
tar czf dss.tar.gz dss

tar解压

//tar xf 要解压的文件名.tar
tar xf dss.tar

//tar xzf 要解压的文件名.tar.gz

tar xzf dss.tar.gz
