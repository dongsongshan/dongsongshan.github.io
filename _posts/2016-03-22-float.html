元素吸顶

// 设置吸顶元素距网页文档顶部的高度
var distance = 1100;
// 屏幕高度
var viewHeight = $(window).height();
// 吸顶元素自身高度
var selfHeight = $('.float').outerHeight();
if ( $(window).scrollTop() < distance-(viewHeight-selfHeight)/2) {
  $('.float').css({
  	"position":"absolute",
  	"top": distance + 'px',
  	"margin-top": ""
  });
} else {
  $('.float').css({
  	"position":"fixed",
  	"top":"50%",
  	"margin-top": (-selfHeight/2) + 'px'
  });	
}
