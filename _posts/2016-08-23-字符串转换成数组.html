字符串转成数组split()

var str = hello world;

str.split(" "); //["hello", "world"]
str.split("");  //["h", "e", "l", "l", "o", " ", "w", "o", "r", "l", "d"]


var str = '13579';
str.split(""); //["1", "3", "5", "7", "9"]

字符串转成数字类型

var str = '13579';
//-------第一种--------
parseInt(str);         // 13579
typeof parseInt(str);  //number
//-------第二种--------
Number(str)            // 13579
//-------第三种--------
function strToInt(s){
  return s*1;
}
strToInt(str)         // 13579

数字类型转换成字符串

var int = 13579;
String(int);         // "13579"
typeof String(int)   // string

数组转换成字符串 join()

join(-)的参数是表示以-分割，没有参数的话，默认以逗号分割

arr = ["h", "e", "l", "l", "o"];

arr.join();       // 'h,e,l,l,o'
arr.join('');     // 'hello'
