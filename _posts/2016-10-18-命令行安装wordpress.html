--- layout: default title: 命令行安装wordpress ---

1,下载

wp core download --locale=zh_CN

2,配置

// 配置数据库信息，wp-config.php
wp core config --dbname=数据库名 --dbuser=用户名 --dbpass=密码

3,安装

wp core install --url=网站的地址 --title=网站名字 --admin_user=管理员名 --admin_password=管理员密码 --admin_email=管理员邮箱

4,设置语言

// 查看语言列表
wp core language list
// 下载中文
wp core language install zh_CN
// 激活中文
wp core language activate zh_CN
// 上面两步也可以一步到位，下载并激活
wp core language install zh_CN --activate
