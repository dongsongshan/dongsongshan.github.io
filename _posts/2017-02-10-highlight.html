--- layout: default title: 代码高亮highlight.js ---

轻轻三部就能让代码高亮显示

  
//引入css
<link href="http://cdn.bootcss.com/highlight.js/9.9.0/styles/atom-one-dark.min.css" rel="stylesheet">

//引入js
<script src="http://cdn.bootcss.com/highlight.js/9.9.0/highlight.min.js">

//初始化highlight
<script>hljs.initHighlightingOnLoad();</script>

  


//如果系统不能识别代码类型 可以显式增加 class="html"
<pre>
  <code class="html">
    <h1>这里是标题</h1>
  </code>
</pre>



// php 代码
echo "Hello World"

禁止高亮


<pre><code class="nohighlight">...</code></pre>
